59 if (poses_.size() < 2) {

66 // Extract translations
67 Eigen::Vector3<DataType> pos_prev = poses_[i-1].template block<3, 1>(0, 3);
68 Eigen::Vector3<DataType> pos_curr = poses_[i].template block<3, 1>(0, 3);
