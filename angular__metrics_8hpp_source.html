144template<typename T>
145class CorrelationMetric : public base_metric_t<CorrelationMetric<T>, T>
146{
147public:
148 using element_type = T;
149
150 constexpr T distance_impl(const T* a, const T* b, std::size_t size) const
151 {
152 if (size == 0) {
153 return T(0);
154 }
155
156 // Compute means
157 T mean_a = std::accumulate(a, a + size, T(0)) / size;
158 T mean_b = std::accumulate(b, b + size, T(0)) / size;
159
160 // Compute correlation
161 T cov {};
162 T var_a {};
163 T var_b {};
164
165 for (std::size_t i = 0; i < size; ++i) {
166 T diff_a = a[i] - mean_a;
167 T diff_b = b[i] - mean_b;
168 cov += diff_a * diff_b;
169 var_a += diff_a * diff_a;
170 var_b += diff_b * diff_b;
171 }
172
173 if (var_a < std::numeric_limits<T>::epsilon() ||
174 var_b < std::numeric_limits<T>::epsilon()) {
175 return T(1); // Maximum distance for constant vectors
176 }
177
178 T correlation = cov / std::sqrt(var_a * var_b);
179
180 // Clamp to handle numerical errors
181 correlation = std::max(T(-1), std::min(T(1), correlation));
182
183 // Convert correlation to distance: (1 - correlation) / 2 gives [0, 1]
184 return (T(1) - correlation) / T(2);
185 }
186
187 constexpr T squared_distance_impl(const T* a, const T* b, std::size_t size) const
188 {
189 T dist = distance_impl(a, b, size);
190 return dist * dist;
191 }
