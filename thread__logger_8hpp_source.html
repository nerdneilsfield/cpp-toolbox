419 template<typename T, typename... Args>
420 auto format_message(const char* format, T&& value, Args&&... args)
421 -> std::string
422 {
423 std::string result;
424 while (*format) {
425 if (*format == '{' && *(format + 1) == '}') {
426 std::stringstream ss;
427 ss << value;
428 result += ss.str();
429 return result
430 + format_message(format + 2, std::forward<Args>(args)...);
431 }
432 result += *format++;
433 }
434 return result;
435 }
