231// For calculable containers
232template<typename InputType>
233[[nodiscard]] auto calculate_minmax_parallel(const InputType& input)
234 -> std::enable_if_t<detail::is_calculable_container_v<InputType>,
235 minmax_t<typename InputType::value_type>>
236{
237 using ItemType = typename InputType::value_type;
238 using ResultType = minmax_t<ItemType>;
239
240 const auto total_size_signed =
241 std::distance(std::begin(input), std::end(input));
242 constexpr size_t sequential_threshold =
243 1024; // Threshold for switching to sequential
244
245 if (total_size_signed <= 0) { // Handle empty range
246 return ResultType(); // Return default (uninitialized)
247 }
248 const size_t total_size = static_cast<size_t>(total_size_signed);
249
250 if (total_size < sequential_threshold) {
251 return calculate_minmax(input); // Fallback to sequential
252 }
253
254 // --- Parallel Execution Logic ---
255 auto& pool = toolbox::concurrent::default_pool();
256 const size_t num_threads = pool.get_thread_count();
257 const size_t hardware_threads =
258 std::max(1u, std::thread::hardware_concurrency());
259 const size_t min_chunk_size = 256;
260 // Determine number of tasks, aiming for reasonable granularity
261 const size_t max_tasks = std::max(
262 static_cast<size_t>(1), std::max(num_threads, hardware_threads) * 4);
263 const size_t chunk_size =
264 std::max(min_chunk_size,
265 static_cast<size_t>(
266 std::ceil(static_cast<double>(total_size) / max_tasks)));
267 size_t num_tasks = (total_size == 0)
268 ? 0
269 : static_cast<size_t>(
270 std::ceil(static_cast<double>(total_size) / chunk_size));
271 if (num_tasks == 0 && total_size > 0)
272 num_tasks = 1; // Ensure at least one task if not empty
273
274 std::vector<std::future<ResultType>> futures;
275 if (num_tasks == 0) {
276 return ResultType();
277 }
278 futures.reserve(num_tasks);
279
280 auto task_lambda = [](auto chunk_begin_it, auto chunk_end_it) -> ResultType
281 {
282 ResultType local_result;
283 if (chunk_begin_it == chunk_end_it) { // Handle empty chunk
284 local_result.initialized_ = false;
285 return local_result;
286 }
287
288 // Use operator+= for the first element and subsequent elements
289 auto it = chunk_begin_it;
290 local_result += *it; // Initialize using operator+=
291 ++it;
292 for (; it != chunk_end_it; ++it) {
293 local_result += *it; // Update using operator+=
294 }
295 return local_result;
296 };
297
298 auto first = std::begin(input);
299 for (size_t i = 0; i < num_tasks; ++i) {
300 size_t start_offset = i * chunk_size;
301 size_t current_chunk_size = std::min(chunk_size, total_size - start_offset);
302
303 if (current_chunk_size == 0)
304 break; // No more elements
305
306 auto chunk_begin = first;
307 std::advance(chunk_begin, start_offset);
308 auto chunk_end = chunk_begin;
309 std::advance(chunk_end, current_chunk_size);
310
311 futures.emplace_back(pool.submit(task_lambda, chunk_begin, chunk_end));
312 }
313
314 // Reduce the results
315 ResultType final_result; // Default constructed (uninitialized)
316 try {
317 for (auto& fut : futures) {
318 ResultType partial_result = fut.get();
319 // Combine using the helper function which handles initialization state
320 final_result = combine_minmax(final_result, partial_result);
321 }
322 } catch (const std::exception& e) {
323 std::cerr << "Exception during parallel minmax reduction: " << e.what()
324 << '\n';
325 throw; // Rethrow for now
326 } catch (...) {
327 std::cerr << "Unknown exception during parallel minmax reduction.\n";
328 throw; // Rethrow for now
329 }
330
331 return final_result;
332}
