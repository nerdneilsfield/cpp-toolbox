284template<typename T>
285auto to_string_value(const T& value) -> std::string
286{
287 // 专门处理 std::string 本身
288 if constexpr (std::is_same_v<T, std::string>) {
289 return value;
290 }
291 // 如果是可迭代容器（且不是 string），就把它当作列表打印
292 else if constexpr (toolbox::traits::is_iterable_v<T>)
293 {
294 std::ostringstream oss;
295 oss << "[";
296 bool first = true;
297 for (auto const& elem : value) {
298 if (!first) {
299 oss << ", ";
300 }
301 oss << to_string_value(elem); // 递归；如果 elem 也是容器会继续走这里
302 first = false;
303 }
304 oss << "]";
305 return oss.str();
306 }
307 // 否则尝试用 operator<< 打印
308 else
309 {
310 std::ostringstream oss;
311 oss << value; // 只有当 stream << T 可用才会编译
312 return oss.str();
313 }
314}

327inline auto align_text(const std::string& text,
328 size_t width,
329 align_t align,
330 [[maybe_unused]] const std::string& padding = " ")
331 -> std::string
332{
333 if (text.length() >= width) {
334 return text;
335 }
336
337 const size_t padding_length = width - text.length();
338
339 switch (align) {
340 case align_t::RIGHT: {
341 return std::string(padding_length, ' ') + text;
342 }
343 case align_t::CENTER: {
344 const size_t left_padding = padding_length / 2;
345 const size_t right_padding = padding_length - left_padding;
346 return std::string(left_padding, ' ') + text
347 + std::string(right_padding, ' ');
348 }
349 case align_t::LEFT:
350 default: {
351 return text + std::string(padding_length, ' ');
352 }
353 }
354}
