303inline auto align_text(const std::string& text,
304 size_t width,
305 align_t align,
306 [[maybe_unused]] const std::string& padding = " ")
307 -> std::string
308{
309 if (text.length() >= width) {
310 return text;
311 }
312
313 const size_t padding_length = width - text.length();
314
315 switch (align) {
316 case align_t::RIGHT: {
317 return std::string(padding_length, ' ') + text;
318 }
319 case align_t::CENTER: {
320 const size_t left_padding = padding_length / 2;
321 const size_t right_padding = padding_length - left_padding;
322 return std::string(left_padding, ' ') + text
323 + std::string(right_padding, ' ');
324 }
325 case align_t::LEFT:
326 default: {
327 return text + std::string(padding_length, ' ');
328 }
329 }

355 void calculate_col_widths() const
356 {
357 size_t num_columns = m_headers.size();

371 for (const auto& row : m_data) {
372 for (size_t i = 0; i < row.size() && i < m_col_widths.size(); ++i) {
373 m_col_widths[i] = std::max(m_col_widths[i], row[i].length());
374 }

421 os << border_v;

429 // 应用颜色/Apply color

432 if (is_header) {
433 colored_cell = color_handler_t::colorize(
434 aligned_cell, m_style.header_fg, m_style.header_bg);

439 } else {
440 colored_cell = aligned_cell;
441 }
442
443 os << m_style.padding << colored_cell << m_style.padding;
444
445 if (m_style.show_border) {
446 os << border_v;
447 } else {
448 os << " ";
449 }
450 }
451 os << "\n";
452 }
