137 || term_str == "screen-256color" || term_str == "tmux"
138 || term_str == "tmux-256color" || term_str == "rxvt"
139 || term_str == "rxvt-unicode" || term_str == "linux"

141#endif

236 static auto colorize(const std::string& text, color_t fg, color_t bg)

282template<typename T>
283auto to_string_value(const T& value) -> std::string
284{
285 // 专门处理 std::string 本身
286 if constexpr (std::is_same_v<T, std::string>) {
287 return value;
288 }
289 // 如果是可迭代容器（且不是 string），就把它当作列表打印
290 else if constexpr (toolbox::traits::is_iterable_v<T>)
291 {
292 std::ostringstream oss;
293 oss << "[";
294 bool first = true;
295 for (auto const& elem : value) {
296 if (!first) {
297 oss << ", ";
298 }
299 oss << to_string_value(elem); // 递归；如果 elem 也是容器会继续走这里
300 first = false;
301 }
302 oss << "]";
303 return oss.str();
304 }
305 // 否则尝试用 operator<< 打印
306 else
307 {
308 std::ostringstream oss;
309 oss << value; // 只有当 stream << T 可用才会编译
310 return oss.str();
311 }
312}

325inline auto align_text(const std::string& text,
326 size_t width,
327 align_t align,
328 [[maybe_unused]] const std::string& padding = " ")
329 -> std::string
330{
331 if (text.length() >= width) {
332 return text;
333 }
334
335 const size_t padding_length = width - text.length();
336
337 switch (align) {
338 case align_t::RIGHT: {
339 return std::string(padding_length, ' ') + text;
340 }
341 case align_t::CENTER: {
342 const size_t left_padding = padding_length / 2;
343 const size_t right_padding = padding_length - left_padding;
344 return std::string(left_padding, ' ') + text
345 + std::string(right_padding, ' ');
346 }
347 case align_t::LEFT:
348 default: {
349 return text + std::string(padding_length, ' ');
350 }
351 }
352}
