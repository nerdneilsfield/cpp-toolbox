230 bool apply_ratio_test(const std::vector<correspondence_t>& candidates) const
231 {
232 // Lowe's ratio test: 检查最佳匹配是否显著优于次佳匹配 / Check if best match is significantly better than second best
233 if (candidates.size() < 2) {
234 return true; // 如果只有一个匹配，默认通过 / If only one match, pass by default
235 }
236
237 return (candidates[0].distance < m_ratio * candidates[1].distance);

260 float compute_geometric_consistency(const std::vector<correspondence_t>& correspondences) const
261 {
262 if (correspondences.size() < 3) {
263 return 0.0f;
264 }
265
266 // 简单的几何一致性检查：比较点对之间的距离比率 / Simple geometric consistency: compare distance ratios between point pairs
267 float consistency_score = 0.0f;
268 std::size_t valid_pairs = 0;
269
270 for (std::size_t i = 0; i < correspondences.size() - 1; ++i) {
271 for (std::size_t j = i + 1; j < correspondences.size(); ++j) {
272 // 计算源点对之间的距离 / Compute distance between source point pair
273 const auto& src_p1 = m_src_cloud->points[correspondences[i].src_idx];
274 const auto& src_p2 = m_src_cloud->points[correspondences[j].src_idx];
275 DataType src_dist = std::sqrt(
276 (src_p1.x - src_p2.x) * (src_p1.x - src_p2.x) +
277 (src_p1.y - src_p2.y) * (src_p1.y - src_p2.y) +
278 (src_p1.z - src_p2.z) * (src_p1.z - src_p2.z));
279
280 // 计算目标点对之间的距离 / Compute distance between target point pair
281 const auto& dst_p1 = m_dst_cloud->points[correspondences[i].dst_idx];
282 const auto& dst_p2 = m_dst_cloud->points[correspondences[j].dst_idx];
283 DataType dst_dist = std::sqrt(
284 (dst_p1.x - dst_p2.x) * (dst_p1.x - dst_p2.x) +
285 (dst_p1.y - dst_p2.y) * (dst_p1.y - dst_p2.y) +
286 (dst_p1.z - dst_p2.z) * (dst_p1.z - dst_p2.z));
287
288 // 避免除零 / Avoid division by zero
289 if (src_dist > 0.001 && dst_dist > 0.001) {
290 DataType ratio = src_dist / dst_dist;
291 // 如果比率接近1，说明几何关系保持良好 / If ratio is close to 1, geometry is well preserved
292 if (ratio > 0.8 && ratio < 1.2) {
293 consistency_score += 1.0f;
294 }
295 valid_pairs++;
296 }
297 }
298 }
299
300 return valid_pairs > 0 ? consistency_score / valid_pairs : 0.0f;
301 }
