177template<typename DataType, typename KNN>
178void shot_extractor_t<DataType, KNN>::compute_local_reference_frame(
179 const point_cloud& cloud,
180 const point_cloud& normals,
181 std::size_t index,
182 const std::vector<std::size_t>& neighbor_indices,
183 local_rf_t& lrf) const
184{
185 const auto& center = cloud.points[index];
186 const auto& normal = normals.points[index];
187
188 // Compute weighted covariance matrix
189 std::vector<data_type> weights(neighbor_indices.size());
190 data_type total_weight = 0;
191
192 for (std::size_t i = 0; i < neighbor_indices.size(); ++i)
193 {
194 point_t<data_type> diff;
195 diff.x = cloud.points[neighbor_indices[i]].x - center.x;
196 diff.y = cloud.points[neighbor_indices[i]].y - center.y;
197 diff.z = cloud.points[neighbor_indices[i]].z - center.z;
198 data_type dist = diff.norm();
199 weights[i] = DataType(1) / (dist + DataType(1e-6));
200 total_weight += weights[i];
201 }
202
203 // Normalize weights
204 for (auto& w : weights)
205 {
206 w /= total_weight;
207 }
208
209 // Compute weighted covariance
210 data_type cov[9];
211 compute_weighted_covariance(cloud, index, neighbor_indices, weights, cov);
212
213 // Eigen decomposition
214 Eigen::Matrix<DataType, 3, 3> cov_matrix;
215 cov_matrix << cov[0], cov[1], cov[2],
216 cov[3], cov[4], cov[5],
217 cov[6], cov[7], cov[8];
218
219 Eigen::SelfAdjointEigenSolver<Eigen::Matrix<DataType, 3, 3>> solver(cov_matrix);
220 auto eigenvectors = solver.eigenvectors();
221
222 // Set z-axis as the normal
223 lrf.z_axis = normal;
224
225 // x-axis is the eigenvector with largest eigenvalue orthogonal to z
226 lrf.x_axis = point_t<data_type>(eigenvectors(0, 2), eigenvectors(1, 2), eigenvectors(2, 2));
227
228 // Make x-axis orthogonal to z-axis
229 data_type dot_product = lrf.x_axis.dot(lrf.z_axis);
230 lrf.x_axis.x = lrf.x_axis.x - lrf.z_axis.x * dot_product;
231 lrf.x_axis.y = lrf.x_axis.y - lrf.z_axis.y * dot_product;
232 lrf.x_axis.z = lrf.x_axis.z - lrf.z_axis.z * dot_product;
233 auto x_normalized = lrf.x_axis.normalize();
234 lrf.x_axis.x = static_cast<data_type>(x_normalized.x);
235 lrf.x_axis.y = static_cast<data_type>(x_normalized.y);
236 lrf.x_axis.z = static_cast<data_type>(x_normalized.z);
237
238 // y-axis = z Ã— x
239 lrf.y_axis = lrf.z_axis.cross(lrf.x_axis);
240 auto y_normalized = lrf.y_axis.normalize();
241 lrf.y_axis.x = static_cast<data_type>(y_normalized.x);
242 lrf.y_axis.y = static_cast<data_type>(y_normalized.y);
243 lrf.y_axis.z = static_cast<data_type>(y_normalized.z);
244
245 // Disambiguate the sign of axes
246 data_type positive_count = 0;
247 for (std::size_t neighbor_idx : neighbor_indices)
248 {
249 point_t<data_type> diff;
250 diff.x = cloud.points[neighbor_idx].x - center.x;
251 diff.y = cloud.points[neighbor_idx].y - center.y;
252 diff.z = cloud.points[neighbor_idx].z - center.z;
253 if (diff.dot(lrf.x_axis) > 0)
254 positive_count++;
255 }
256
257 if (positive_count < neighbor_indices.size() / 2)
258 {
259 lrf.x_axis.x = -lrf.x_axis.x;
260 lrf.x_axis.y = -lrf.x_axis.y;
261 lrf.x_axis.z = -lrf.x_axis.z;
262 lrf.y_axis.x = -lrf.y_axis.x;
263 lrf.y_axis.y = -lrf.y_axis.y;
264 lrf.y_axis.z = -lrf.y_axis.z;
265 }
266}

268template<typename DataType, typename KNN>
269void shot_extractor_t<DataType, KNN>::compute_shot_feature(
270 const point_cloud& cloud,
271 const point_cloud& normals,
272 std::size_t index,
273 const std::vector<std::size_t>& neighbor_indices,
274 const local_rf_t& lrf,
275 signature_type& shot) const
276{
277 // Initialize histogram
278 std::fill(shot.histogram.begin(), shot.histogram.end(), DataType(0));
279
280 const auto& center = cloud.points[index];
281 constexpr std::size_t num_spatial_bins = 32;
282 constexpr std::size_t num_angular_bins = 11;
283
284 // Process each neighbor
285 for (std::size_t neighbor_idx : neighbor_indices)
286 {
287 if (neighbor_idx == index) continue;
288
289 const auto& neighbor_point = cloud.points[neighbor_idx];
290 const auto& neighbor_normal = normals.points[neighbor_idx];
291
292 // Compute spatial bin
293 std::size_t spatial_bin = compute_spatial_bin(neighbor_point, center, lrf, m_search_radius);
294 if (spatial_bin >= num_spatial_bins) continue;
295
296 // Compute angular bin
297 std::size_t angular_bin = compute_angular_bin(neighbor_normal, lrf);
298 if (angular_bin >= num_angular_bins) continue;
299
300 // Add to histogram
301 std::size_t hist_idx = spatial_bin * num_angular_bins + angular_bin;
302 point_t<data_type> diff;
303 diff.x = neighbor_point.x - center.x;
304 diff.y = neighbor_point.y - center.y;
305 diff.z = neighbor_point.z - center.z;
306 data_type distance = diff.norm();
307 data_type weight = DataType(1) - (distance / m_search_radius);
308 shot.histogram[hist_idx] += weight;
309 }
310
311 // Normalize histogram
312 data_type norm = 0;
313 for (std::size_t i = 0; i < signature_type::HISTOGRAM_SIZE; ++i)
314 {
315 norm += shot.histogram[i] * shot.histogram[i];
316 }
317
318 if (norm > DataType(1e-6))
319 {
320 norm = DataType(1) / std::sqrt(norm);
321 for (std::size_t i = 0; i < signature_type::HISTOGRAM_SIZE; ++i)
322 {
323 shot.histogram[i] *= norm;
324 }
325 }

328template<typename DataType, typename KNN>
329void shot_extractor_t<DataType, KNN>::compute_weighted_covariance(
330 const point_cloud& cloud,
331 std::size_t center_idx,
332 const std::vector<std::size_t>& indices,
333 const std::vector<data_type>& weights,
334 data_type cov[9]) const
335{
336 const auto& center = cloud.points[center_idx];
337
338 // Initialize covariance matrix
339 std::fill(cov, cov + 9, DataType(0));
340
341 // Compute weighted mean
342 point_t<data_type> mean(0, 0, 0);
343 for (std::size_t i = 0; i < indices.size(); ++i)
344 {
345 point_t<data_type> diff;
346 diff.x = cloud.points[indices[i]].x - center.x;
347 diff.y = cloud.points[indices[i]].y - center.y;
348 diff.z = cloud.points[indices[i]].z - center.z;
349 mean.x += weights[i] * diff.x;
350 mean.y += weights[i] * diff.y;
351 mean.z += weights[i] * diff.z;
352 }
353
354 // Compute covariance
355 for (std::size_t i = 0; i < indices.size(); ++i)
356 {
357 point_t<data_type> diff;
358 diff.x = cloud.points[indices[i]].x - center.x - mean.x;
359 diff.y = cloud.points[indices[i]].y - center.y - mean.y;
360 diff.z = cloud.points[indices[i]].z - center.z - mean.z;
361 data_type w = weights[i];
362
363 cov[0] += w * diff.x * diff.x;
364 cov[1] += w * diff.x * diff.y;
365 cov[2] += w * diff.x * diff.z;
366 cov[3] += w * diff.y * diff.x;
367 cov[4] += w * diff.y * diff.y;
368 cov[5] += w * diff.y * diff.z;
369 cov[6] += w * diff.z * diff.x;
370 cov[7] += w * diff.z * diff.y;
371 cov[8] += w * diff.z * diff.z;
372 }

375template<typename DataType, typename KNN>
376std::size_t shot_extractor_t<DataType, KNN>::compute_spatial_bin(
377 const point_t<data_type>& point,
378 const point_t<data_type>& center,
379 const local_rf_t& lrf,
380 data_type radius) const
381{
382 point_t<data_type> local_point;
383 local_point.x = point.x - center.x;
384 local_point.y = point.y - center.y;
385 local_point.z = point.z - center.z;
386
387 // Transform to local reference frame
388 data_type x = local_point.dot(lrf.x_axis);
389 data_type y = local_point.dot(lrf.y_axis);
390 data_type z = local_point.dot(lrf.z_axis);
391
392 // Compute spherical coordinates
393 data_type r = std::sqrt(x*x + y*y + z*z);
394 data_type theta = std::atan2(y, x); // Azimuth angle
395 data_type phi = std::acos(z / (r + DataType(1e-6))); // Elevation angle
396
397 // Normalize coordinates
398 r = r / radius;
399 theta = (theta + DataType(M_PI)) / (DataType(2) * DataType(M_PI));
400 phi = phi / DataType(M_PI);
401
402 // Determine spatial bin (2 radial Ã— 4 azimuth Ã— 4 elevation = 32 bins)
403 std::size_t r_bin = (r < DataType(0.5)) ? 0 : 1;
404 std::size_t theta_bin = static_cast<std::size_t>(theta * 4);
405 std::size_t phi_bin = static_cast<std::size_t>(phi * 4);
406
407 // Clamp bins
408 theta_bin = std::min(theta_bin, std::size_t(3));
409 phi_bin = std::min(phi_bin, std::size_t(3));
410
411 return r_bin * 16 + theta_bin * 4 + phi_bin;
412}

414template<typename DataType, typename KNN>
415std::size_t shot_extractor_t<DataType, KNN>::compute_angular_bin(
416 const point_t<data_type>& normal,
417 const local_rf_t& lrf) const
418{
419 // Project normal onto local reference frame
420 data_type cos_angle = normal.dot(lrf.z_axis);
421
422 // Clamp to [-1, 1]
423 cos_angle = std::max(DataType(-1), std::min(DataType(1), cos_angle));
424
425 // Convert to bin index (11 bins from -1 to 1)
426 data_type normalized = (cos_angle + DataType(1)) / DataType(2);
427 std::size_t bin = static_cast<std::size_t>(normalized * 11);
428
429 return std::min(bin, std::size_t(10));
430}
