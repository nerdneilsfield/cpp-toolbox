58template<typename Iterator, typename Function>
59CPP_TOOLBOX_EXPORT void parallel_for_each(Iterator begin,
60 Iterator end,
61 Function func)
62{
63 using traits = std::iterator_traits<Iterator>;
64 static_assert(std::is_base_of<std::random_access_iterator_tag,
65 typename traits::iterator_category>::value,
66 "parallel_for_each currently requires random access iterators "
67 "for efficient chunking.");
68
69 const auto total_size = std::distance(begin, end);
70 if (total_size <= 0) {
71 return;
72 }
73
74 auto& pool = default_pool();
75 const size_t num_threads = pool.get_thread_count();
76 const size_t min_chunk_size = 1;
77 const size_t hardware_threads =
78 std::max(1u, std::thread::hardware_concurrency());
79 const size_t num_tasks = std::max(num_threads, hardware_threads);
80
81 size_t chunk_size =
82 std::max(min_chunk_size,
83 static_cast<size_t>(
84 std::ceil(static_cast<double>(total_size) / num_tasks)));
85
86 std::vector<std::future<void>> futures;
87 futures.reserve(num_tasks);
88
89 Iterator chunk_begin = begin;
90 for (size_t i = 0; i < num_tasks; ++i) {
91 Iterator chunk_end = chunk_begin;
92 size_t current_chunk_actual_size = std::min(
93 chunk_size, static_cast<size_t>(std::distance(chunk_begin, end)));
94 if (current_chunk_actual_size == 0)
95 break;
96 std::advance(chunk_end, current_chunk_actual_size);
97
98 futures.emplace_back(pool.submit(
99 [chunk_begin, chunk_end, func]() mutable
100 {
101 for (auto it = chunk_begin; it != chunk_end; ++it) {
102 func(*it);
103 }
104 }));
105
106 chunk_begin = chunk_end;
107
108 if (chunk_begin == end) {
109 break;
110 }
111 }
112
113 try {
114 for (auto& fut : futures) {
115 fut.get();
116 }
117 } catch (...) {
118 throw;
119 }
120}

206template<typename InputIt, typename OutputIt, typename UnaryOperation>
207CPP_TOOLBOX_EXPORT void parallel_transform(InputIt first1,
208 InputIt last1,
209 OutputIt d_first,
210 UnaryOperation unary_op)
211{
212 using InputTraits = std::iterator_traits<InputIt>;
213 using OutputTraits = std::iterator_traits<OutputIt>;
214
215 static_assert(
216 std::is_base_of<std::random_access_iterator_tag,
217 typename InputTraits::iterator_category>::value,
218 "parallel_transform currently requires random access input iterators.");
219 static_assert(
220 std::is_base_of<std::random_access_iterator_tag,
221 typename OutputTraits::iterator_category>::value,
222 "parallel_transform currently requires random access output iterators.");
223
224 const auto total_size = std::distance(first1, last1);
225 if (total_size <= 0) {
226 return;
227 }
228
229 auto& pool = default_pool();
230 const size_t num_threads = pool.get_thread_count();
231 const size_t min_chunk_size = 1;
232 const size_t hardware_threads =
233 std::max(1u, std::thread::hardware_concurrency());

235
236 size_t chunk_size =
237 std::max(min_chunk_size,
238 static_cast<size_t>(
239 std::ceil(static_cast<double>(total_size) / num_tasks)));
240
241 std::vector<std::future<void>> futures;
242 futures.reserve(num_tasks);
243
244 size_t chunk_start_index = 0;
245 for (size_t i = 0; i < num_tasks; ++i) {
246 size_t chunk_end_index = std::min(chunk_start_index + chunk_size,
247 static_cast<size_t>(total_size));

249 if (chunk_start_index >= chunk_end_index)
250 break;

257 for (size_t k = chunk_start_index; k < chunk_end_index; ++k) {
258 *(d_first + static_cast<diff_t>(k)) =

260 }
261 }));
262
263 chunk_start_index = chunk_end_index;
264 if (chunk_start_index >= static_cast<size_t>(total_size))
265 break;
266 }
267
268 try {
269 for (auto& fut : futures) {
270 fut.get();
271 }
272 } catch (...) {
273 throw;
274 }

364template<typename Iterator, typename T, typename BinaryOperation>
365CPP_TOOLBOX_EXPORT T parallel_reduce(Iterator begin,
366 Iterator end,
367 T identity,
368 BinaryOperation reduce_op)
369{
370 using traits = std::iterator_traits<Iterator>;
371 static_assert(std::is_base_of<std::random_access_iterator_tag,
372 typename traits::iterator_category>::value,
373 "parallel_reduce currently requires random access iterators "
374 "for efficient chunking.");
375
376 const auto total_size = std::distance(begin, end);
377 if (total_size <= 0) {
378 return identity;
379 }
380
381 auto& pool = default_pool();
382 const size_t num_threads = pool.get_thread_count();
383 const size_t min_chunk_size = 256;
384 const size_t hardware_threads =
385 std::max(1u, std::thread::hardware_concurrency());
386 const size_t num_tasks = std::max(static_cast<size_t>(1ul),
387 std::min(num_threads, hardware_threads));
388
389 size_t chunk_size =
390 std::max(min_chunk_size,
391 static_cast<size_t>(
392 std::ceil(static_cast<double>(total_size) / num_tasks)));
393 size_t actual_num_tasks = static_cast<size_t>(
394 std::ceil(static_cast<double>(total_size) / chunk_size));
395
396 std::vector<std::future<T>> futures;
397 futures.reserve(actual_num_tasks);
398
399 Iterator chunk_begin = begin;
400 for (size_t i = 0; i < actual_num_tasks; ++i) {
401 Iterator chunk_end = chunk_begin;
402 size_t current_chunk_actual_size = std::min(
403 chunk_size, static_cast<size_t>(std::distance(chunk_begin, end)));
404 if (current_chunk_actual_size == 0)
405 break;
406 std::advance(chunk_end, current_chunk_actual_size);
407
408 futures.emplace_back(pool.submit(
409 [chunk_begin, chunk_end, reduce_op, identity]() -> T
410 {
411 if (chunk_begin == chunk_end) {
412 return identity;
413 }
414
415 auto it = chunk_begin;
416 T local_result = *it;
417 ++it;
418
419 for (; it != chunk_end; ++it) {
420 local_result = reduce_op(local_result, *it);
421 }
422 return local_result;
423 }));
424
425 chunk_begin = chunk_end;
426 if (chunk_begin == end)
427 break;
428 }
429
430 std::vector<T> partial_results;
431 partial_results.reserve(futures.size());
432
433 try {
434 for (auto& fut : futures) {
435 partial_results.push_back(fut.get());
436 }
437
438 if (partial_results.empty()) {
439 return identity;
440 }
441
442 while (partial_results.size() > 1) {
443 size_t pair_count = partial_results.size() / 2;
444 std::vector<std::future<T>> merge_futs;
445 merge_futs.reserve(pair_count);
446
447 for (size_t i = 0; i < pair_count; ++i) {
448 T lhs = partial_results[2 * i];
449 T rhs = partial_results[2 * i + 1];
450 merge_futs.emplace_back(pool.submit([lhs, rhs, reduce_op]()
451 { return reduce_op(lhs, rhs); }));
452 }
453
454 if (partial_results.size() % 2 == 1) {
455 partial_results[pair_count] = partial_results.back();
456 }
457 partial_results.resize(pair_count + (partial_results.size() % 2));
458
459 for (size_t i = 0; i < merge_futs.size(); ++i) {
460 partial_results[i] = merge_futs[i].get();
461 }
462 }
463 } catch (...) {
464 throw;
465 }
466
467 return reduce_op(identity, partial_results.front());
468}

535template<typename InputIt,
536 typename OutputIt,
537 typename T,
538 typename BinaryOperation>
539CPP_TOOLBOX_EXPORT void parallel_inclusive_scan(InputIt first,
540 InputIt last,
541 OutputIt d_first,
542 T init,
543 BinaryOperation binary_op,
544 T identity = T {})
545{
546 using traits = std::iterator_traits<InputIt>;
547 static_assert(std::is_base_of_v<std::random_access_iterator_tag,
548 typename traits::iterator_category>,
549 "parallel_inclusive_scan requires random access iterators.");
550
551 const auto total_size = std::distance(first, last);
552 if (total_size <= 0) {
553 return;
554 }
555
556 auto& pool = default_pool();
557 const size_t num_threads = pool.get_thread_count();
558 const size_t hardware_threads =
559 std::max(1u, std::thread::hardware_concurrency());
560 size_t num_tasks = std::max(num_threads, hardware_threads);
561
562 size_t chunk_size = static_cast<size_t>(
563 std::ceil(static_cast<double>(total_size) / num_tasks));
564 num_tasks = static_cast<size_t>(
565 std::ceil(static_cast<double>(total_size) / chunk_size));
566
567 std::vector<T> chunk_sums(num_tasks);
568 std::vector<std::pair<size_t, size_t>> ranges;
569 ranges.reserve(num_tasks);
570
571 std::vector<std::future<T>> sum_futures;
572 sum_futures.reserve(num_tasks);
573
574 InputIt chunk_begin = first;
575 for (size_t i = 0; i < num_tasks; ++i) {
576 InputIt chunk_end = chunk_begin;
577 size_t current_size = std::min(
578 chunk_size, static_cast<size_t>(std::distance(chunk_begin, last)));
579 if (current_size == 0)
580 break;
581 std::advance(chunk_end, current_size);
582 ranges.emplace_back(static_cast<size_t>(std::distance(first, chunk_begin)),
583 current_size);
584
585 sum_futures.emplace_back(pool.submit(
586 [chunk_begin, chunk_end, identity, binary_op]()
587 {
588 T local_sum = identity;
589 for (auto it = chunk_begin; it != chunk_end; ++it) {
590 local_sum = binary_op(local_sum, *it);
591 }
592 return local_sum;
593 }));
594
595 chunk_begin = chunk_end;
596 if (chunk_begin == last)
597 break;
598 }
599
600 size_t actual_tasks = ranges.size();
601 for (size_t i = 0; i < actual_tasks; ++i) {
602 chunk_sums[i] = sum_futures[i].get();
603 }
604
605 std::vector<T> offsets(actual_tasks);
606 T running = init;
607 for (size_t i = 0; i < actual_tasks; ++i) {
608 offsets[i] = running;
609 running = binary_op(running, chunk_sums[i]);
610 }
611
612 std::vector<std::future<void>> futures;
613 futures.reserve(actual_tasks);
614 for (size_t i = 0; i < actual_tasks; ++i) {
615 size_t start_index = ranges[i].first;
616 size_t len = ranges[i].second;
617
618 InputIt cb =
619 first + static_cast<typename traits::difference_type>(start_index);
620 OutputIt db =
621 d_first + static_cast<typename traits::difference_type>(start_index);
622 T offset = offsets[i];
623
624 futures.emplace_back(pool.submit(
625 [cb, db, len, offset, binary_op]() mutable
626 {
627 T local = offset;
628 for (size_t j = 0; j < len; ++j) {
629 local = binary_op(
630 local,
631 *(cb + static_cast<typename traits::difference_type>(j)));
632 *(db + static_cast<typename traits::difference_type>(j)) = local;
633 }
634 }));
635 }
636
637 for (auto& fut : futures) {
638 fut.get();
639 }
640}

671template<typename RandomIt, typename Compare = std::less<>>
672CPP_TOOLBOX_EXPORT void parallel_merge_sort(RandomIt begin,
673 RandomIt end,
674 Compare comp = Compare())
675{
676 const auto total_size = std::distance(begin, end);
677 if (total_size <= 1) {
678 return;
679 }
680
681 auto& pool = default_pool();
682 const size_t num_threads = pool.get_thread_count();
683 const size_t hardware_threads =
684 std::max(1u, std::thread::hardware_concurrency());
685 size_t num_tasks = std::max(num_threads, hardware_threads);
686
687 size_t chunk_size = static_cast<size_t>(
688 std::ceil(static_cast<double>(total_size) / num_tasks));
689 num_tasks = static_cast<size_t>(
690 std::ceil(static_cast<double>(total_size) / chunk_size));
691
692 std::vector<std::future<void>> futures;
693 futures.reserve(num_tasks);
694 std::vector<std::pair<RandomIt, RandomIt>> ranges;
695 ranges.reserve(num_tasks);
696
697 RandomIt chunk_begin = begin;
698 for (size_t i = 0; i < num_tasks; ++i) {
699 RandomIt chunk_end = chunk_begin;
700 size_t current_size = std::min(
701 chunk_size, static_cast<size_t>(std::distance(chunk_begin, end)));
702 if (current_size == 0)
703 break;
704 std::advance(chunk_end, current_size);
705 ranges.emplace_back(chunk_begin, chunk_end);
706 futures.emplace_back(
707 pool.submit([chunk_begin, chunk_end, comp]()
708 { std::sort(chunk_begin, chunk_end, comp); }));
709 chunk_begin = chunk_end;
710 if (chunk_begin == end)
711 break;
712 }
713
714 for (auto& fut : futures) {
715 fut.get();
716 }
717
718 while (ranges.size() > 1) {
719 std::vector<std::pair<RandomIt, RandomIt>> new_ranges;
720 std::vector<std::future<void>> merge_futs;
721 for (size_t i = 0; i + 1 < ranges.size(); i += 2) {
722 auto begin1 = ranges[i].first;
723 auto mid = ranges[i].second;
724 auto end2 = ranges[i + 1].second;
725 merge_futs.emplace_back(pool.submit(
726 [begin1, mid, end2, comp]() mutable
727 { std::inplace_merge(begin1, mid, end2, comp); }));
728 new_ranges.emplace_back(begin1, end2);
729 }
730
731 for (auto& fut : merge_futs) {
732 fut.get();
733 }
734
735 if (ranges.size() % 2 == 1) {
736 new_ranges.push_back(ranges.back());
737 }
738 ranges.swap(new_ranges);
739 }
740}
