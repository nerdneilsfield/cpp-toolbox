58template<typename Iterator, typename Function>
59CPP_TOOLBOX_EXPORT void parallel_for_each(Iterator begin,
60 Iterator end,
61 Function func)
62{
63 using traits = std::iterator_traits<Iterator>;
64 static_assert(std::is_base_of<std::random_access_iterator_tag,
65 typename traits::iterator_category>::value,
66 "parallel_for_each currently requires random access iterators "
67 "for efficient chunking.");
68
69 const auto total_size = std::distance(begin, end);
70 if (total_size <= 0) {
71 return;
72 }
73
74 auto& pool = default_pool();
75 const size_t num_threads = pool.get_thread_count();
76 const size_t min_chunk_size = 1;
77 const size_t hardware_threads =
78 std::max(1u, std::thread::hardware_concurrency());
79 const size_t max_tasks = std::max(num_threads, hardware_threads) * 4;
80
81 size_t chunk_size =
82 std::max(min_chunk_size,
83 static_cast<size_t>(
84 std::ceil(static_cast<double>(total_size) / max_tasks)));
85 size_t num_tasks = static_cast<size_t>(
86 std::ceil(static_cast<double>(total_size) / chunk_size));
87
88 std::vector<std::future<void>> futures;
89 futures.reserve(num_tasks);
90
91 Iterator chunk_begin = begin;
92 for (size_t i = 0; i < num_tasks; ++i) {
93 Iterator chunk_end = chunk_begin;
94 size_t current_chunk_actual_size = std::min(
95 chunk_size, static_cast<size_t>(std::distance(chunk_begin, end)));
96 if (current_chunk_actual_size == 0)
97 break;
98 std::advance(chunk_end, current_chunk_actual_size);
99
100 futures.emplace_back(pool.submit(
101 [chunk_begin, chunk_end, func]() mutable
102 {
103 for (auto it = chunk_begin; it != chunk_end; ++it) {
104 func(*it);
105 }
106 }));
107
108 chunk_begin = chunk_end;
109
110 if (chunk_begin == end) {
111 break;
112 }
113 }
114
115 try {
116 for (auto& fut : futures) {
117 fut.get();
118 }
119 } catch (...) {
120 throw;
121 }
122}

208template<typename InputIt, typename OutputIt, typename UnaryOperation>
209CPP_TOOLBOX_EXPORT void parallel_transform(InputIt first1,
210 InputIt last1,
211 OutputIt d_first,
212 UnaryOperation unary_op)
213{
214 using InputTraits = std::iterator_traits<InputIt>;
215 using OutputTraits = std::iterator_traits<OutputIt>;
216
217 static_assert(
218 std::is_base_of<std::random_access_iterator_tag,
219 typename InputTraits::iterator_category>::value,
220 "parallel_transform currently requires random access input iterators.");
221 static_assert(
222 std::is_base_of<std::random_access_iterator_tag,
223 typename OutputTraits::iterator_category>::value,
224 "parallel_transform currently requires random access output iterators.");
225
226 const auto total_size = std::distance(first1, last1);
227 if (total_size <= 0) {
228 return;
229 }
230
231 auto& pool = default_pool();
232 const size_t num_threads = pool.get_thread_count();
233 const size_t min_chunk_size = 1;
234 const size_t hardware_threads =
235 std::max(1u, std::thread::hardware_concurrency());
236 const size_t max_tasks = std::max(num_threads, hardware_threads) * 4;
237
238 size_t chunk_size =
239 std::max(min_chunk_size,
240 static_cast<size_t>(
241 std::ceil(static_cast<double>(total_size) / max_tasks)));
242 size_t num_tasks = static_cast<size_t>(
243 std::ceil(static_cast<double>(total_size) / chunk_size));
244
245 std::vector<std::future<void>> futures;
246 futures.reserve(num_tasks);
247
248 size_t chunk_start_index = 0;
249 for (size_t i = 0; i < num_tasks; ++i) {
250 size_t chunk_end_index = std::min(chunk_start_index + chunk_size,
251 static_cast<size_t>(total_size));
252
253 if (chunk_start_index >= chunk_end_index)
254 break;
255
256 futures.emplace_back(pool.submit(
257 [first1, d_first, chunk_start_index, chunk_end_index, unary_op]()
258 {
259 using diff_t =
260 typename std::iterator_traits<InputIt>::difference_type;
261 for (size_t k = chunk_start_index; k < chunk_end_index; ++k) {
262 *(d_first + static_cast<diff_t>(k)) =
263 unary_op(*(first1 + static_cast<diff_t>(k)));
264 }
265 }));
266
267 chunk_start_index = chunk_end_index;
268 if (chunk_start_index >= static_cast<size_t>(total_size))
269 break;
270 }
271
272 try {
273 for (auto& fut : futures) {
274 fut.get();
275 }
276 } catch (...) {
277 throw;
278 }
279}

368template<typename Iterator, typename T, typename BinaryOperation>
369CPP_TOOLBOX_EXPORT T parallel_reduce(Iterator begin,
370 Iterator end,
371 T identity,
372 BinaryOperation reduce_op)
373{
374 using traits = std::iterator_traits<Iterator>;
375 static_assert(std::is_base_of<std::random_access_iterator_tag,
376 typename traits::iterator_category>::value,
377 "parallel_reduce currently requires random access iterators "
378 "for efficient chunking.");
379
380 const auto total_size = std::distance(begin, end);
381 if (total_size <= 0) {
382 return identity;
383 }
384
385 auto& pool = default_pool();
386 const size_t num_threads = pool.get_thread_count();
387 const size_t min_chunk_size = 256;
388 const size_t hardware_threads =
389 std::max(1u, std::thread::hardware_concurrency());
390 const size_t num_tasks = std::max(static_cast<size_t>(1ul),
391 std::min(num_threads, hardware_threads));
392
393 size_t chunk_size =
394 std::max(min_chunk_size,
395 static_cast<size_t>(
396 std::ceil(static_cast<double>(total_size) / num_tasks)));
397 size_t actual_num_tasks = static_cast<size_t>(
398 std::ceil(static_cast<double>(total_size) / chunk_size));
399
400 std::vector<std::future<T>> futures;
401 futures.reserve(actual_num_tasks);
402
403 Iterator chunk_begin = begin;
404 for (size_t i = 0; i < actual_num_tasks; ++i) {
405 Iterator chunk_end = chunk_begin;
406 size_t current_chunk_actual_size = std::min(
407 chunk_size, static_cast<size_t>(std::distance(chunk_begin, end)));
408 if (current_chunk_actual_size == 0)
409 break;
410 std::advance(chunk_end, current_chunk_actual_size);
411
412 futures.emplace_back(pool.submit(
413 [chunk_begin, chunk_end, reduce_op, identity]() -> T
414 {
415 if (chunk_begin == chunk_end) {
416 return identity;
417 }
418
419 auto it = chunk_begin;
420 T local_result = *it;
421 ++it;
422
423 for (; it != chunk_end; ++it) {
424 local_result = reduce_op(local_result, *it);
425 }
426 return local_result;
427 }));
428
429 chunk_begin = chunk_end;
430 if (chunk_begin == end)
431 break;
432 }
433
434 T final_result = identity;
435 try {
436 for (auto& fut : futures) {
437 T partial_result = fut.get();
438 final_result = reduce_op(final_result, partial_result);
439 }
440 } catch (...) {
441 throw;
442 }
443
444 return final_result;
445}
