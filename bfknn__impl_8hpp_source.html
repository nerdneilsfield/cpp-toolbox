33template<typename DataType>
34DataType bfknn_t<DataType>::compute_distance(
35 const point_t<data_type>& p1,
36 const point_t<data_type>& p2,
37 metric_type_t metric) const
38{
39 switch (metric)
40 {
41 case metric_type_t::euclidean:
42 {
43 data_type dx = p1.x - p2.x;
44 data_type dy = p1.y - p2.y;
45 data_type dz = p1.z - p2.z;
46 return std::sqrt(dx * dx + dy * dy + dz * dz);
47 }
48 case metric_type_t::manhattan:
49 {
50 return std::abs(p1.x - p2.x) + std::abs(p1.y - p2.y) + std::abs(p1.z - p2.z);
51 }
52 case metric_type_t::chebyshev:
53 {
54 return std::max({std::abs(p1.x - p2.x), std::abs(p1.y - p2.y), std::abs(p1.z - p2.z)});
55 }
56 case metric_type_t::minkowski:
57 {
58 data_type p = 3.0; // Default p value for Minkowski
59 data_type sum = std::pow(std::abs(p1.x - p2.x), p) +
60 std::pow(std::abs(p1.y - p2.y), p) +
61 std::pow(std::abs(p1.z - p2.z), p);
62 return std::pow(sum, 1.0 / p);
63 }
64 default:
65 return compute_distance(p1, p2, metric_type_t::euclidean);
66 }
67}

69template<typename DataType>
70bool bfknn_t<DataType>::kneighbors_impl(
71 const point_t<data_type>& query_point,
72 std::size_t num_neighbors,
73 std::vector<std::size_t>& indices,
74 std::vector<data_type>& distances)
75{
76 if (!m_cloud || m_cloud->empty())
77 {
78 return false;
79 }
80
81 const std::size_t cloud_size = m_cloud->size();
82 num_neighbors = std::min(num_neighbors, cloud_size);
83
84 // Compute all distances
85 std::vector<std::pair<data_type, std::size_t>> distance_index_pairs;
86 distance_index_pairs.reserve(cloud_size);
87
88 for (std::size_t i = 0; i < cloud_size; ++i)
89 {
90 data_type dist = compute_distance(query_point, m_cloud->points[i], m_metric);
91 distance_index_pairs.emplace_back(dist, i);
92 }
93
94 // Partial sort to get k nearest neighbors
95 std::partial_sort(distance_index_pairs.begin(),
96 distance_index_pairs.begin() + num_neighbors,
97 distance_index_pairs.end(),
98 [](const auto& a, const auto& b) { return a.first < b.first; });
99
100 // Extract results
101 indices.resize(num_neighbors);
102 distances.resize(num_neighbors);
103 for (std::size_t i = 0; i < num_neighbors; ++i)
104 {
105 distances[i] = distance_index_pairs[i].first;
106 indices[i] = distance_index_pairs[i].second;
107 }
108
109 return true;
110}

142
143 // Extract results
144 indices.reserve(distance_index_pairs.size());
145 distances.reserve(distance_index_pairs.size());
146 for (const auto& [dist, idx] : distance_index_pairs)

148 distances.push_back(dist);

150 }
151
