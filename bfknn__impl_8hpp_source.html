49template<typename Element, typename Metric>
50bool bfknn_generic_t<Element, Metric>::kneighbors_impl(
51 const element_type& query,
52 std::size_t num_neighbors,
53 std::vector<std::size_t>& indices,
54 std::vector<distance_type>& distances)
55{
56 if (!m_data || m_data->empty())
57 {
58 return false;
59 }
60
61 const std::size_t data_size = m_data->size();
62 num_neighbors = std::min(num_neighbors, data_size);
63
64 // Compute all distances
65 std::vector<std::pair<distance_type, std::size_t>> distance_index_pairs;
66 distance_index_pairs.reserve(data_size);
67
68 for (std::size_t i = 0; i < data_size; ++i)
69 {
70 distance_type dist;
71 if (m_use_runtime_metric && m_runtime_metric)
72 {
73 if constexpr (std::is_same_v<Element, toolbox::types::point_t<typename Element::value_type>>) {
74 auto adapter = std::static_pointer_cast<toolbox::metrics::PointMetricAdapter<typename Element::value_type>>(m_runtime_metric);
75 dist = (*adapter)(query, (*m_data)[i]);
76 } else {
77 dist = (*m_runtime_metric)(query, (*m_data)[i]);
78 }
79 }
80 else
81 {
82 dist = m_compile_time_metric(query, (*m_data)[i]);
83 }
84 distance_index_pairs.emplace_back(dist, i);
85 }
86
87 // Partial sort to get k nearest neighbors
88 std::partial_sort(distance_index_pairs.begin(),
89 distance_index_pairs.begin() + num_neighbors,
90 distance_index_pairs.end(),
91 [](const auto& a, const auto& b) { return a.first < b.first; });
92
93 // Extract results
94 indices.resize(num_neighbors);
95 distances.resize(num_neighbors);
96 for (std::size_t i = 0; i < num_neighbors; ++i)
97 {
98 distances[i] = distance_index_pairs[i].first;
99 indices[i] = distance_index_pairs[i].second;
100 }
101
102 return true;
103}

133 }
134 }

137 dist = m_compile_time_metric(query, (*m_data)[i]);
138 }
