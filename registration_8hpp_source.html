130template<typename DataType>
131registration_result_t<DataType> quick_registration(

133 const std::shared_ptr<toolbox::types::point_cloud_t<DataType>>& target,

137 registration_result_t<DataType> result;

147 reg.align(result);

149 super_four_pcs_registration_t<DataType> reg;

152 reg.set_overlap(overlap);
153 reg.set_delta(static_cast<DataType>(0.02));
154 reg.enable_smart_indexing(true);
155 reg.align(result);
156 } else {
157 LOG_ERROR_S << "未知算法 / Unknown algorithm: " << algorithm;
158 }
159
160 return result;
