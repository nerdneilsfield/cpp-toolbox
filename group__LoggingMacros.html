Define documentation

#define LOG_TRACE_F

Macro for TRACE level format logging.

LOG_TRACE_F("Processing item {}", item_id);

#define LOG_DEBUG_F

Macro for DEBUG level format logging.

LOG_DEBUG_F("Received {} bytes of data", data_size);

#define LOG_INFO_F

Macro for INFO level format logging.

LOG_INFO_F("Application started successfully");

#define LOG_WARN_F

Macro for WARN level format logging.

LOG_WARN_F("Disk space is low: {}MB remaining", free_space);

#define LOG_ERROR_F

Macro for ERROR level format logging.

LOG_ERROR_F("Failed to process request: {}", error_message);

#define LOG_CRITICAL_F

Macro for CRITICAL level format logging.

LOG_CRITICAL_F("System failure in module {}", module_name);

#define LOG_TRACE_S

Macro for TRACE level stream logging.

LOG_TRACE_S << "Entering function " << function_name;

#define LOG_DEBUG_S

Macro for DEBUG level stream logging.

LOG_DEBUG_S << "Processing " << items.size() << " items";

#define LOG_INFO_S

Macro for INFO level stream logging.

LOG_INFO_S << "Application version: " << version;

#define LOG_WARN_S

Macro for WARN level stream logging.

LOG_WARN_S << "Low memory: " << free_memory << "MB available";

#define LOG_ERROR_S

Macro for ERROR level stream logging.

LOG_ERROR_S << "Failed to open file: " << filename;

#define LOG_CRITICAL_S

Macro for CRITICAL level stream logging.

LOG_CRITICAL_S << "System shutdown due to critical error in " << component;
