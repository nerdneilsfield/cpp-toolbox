164auto thread_pool_singleton_t::submit(F&& f, Args&&... args)
165 -> std::future<typename std::invoke_result_t<F, Args...>>
166{
167 using return_type = typename std::invoke_result_t<F, Args...>;
168
169 if (stop_.load(std::memory_order_relaxed)) {
170 throw std::runtime_error("Cannot submit task to stopped thread pool");
171 }
172
173 auto task = std::make_shared<std::packaged_task<return_type()>>(
174 std::bind(std::forward<F>(f), std::forward<Args>(args)...));
175
176 std::future<return_type> res = task->get_future();
177
178 tasks_.enqueue(
179 [task]() mutable
180 {
181 try {
182 (*task)();
183 } catch (const std::future_error& fe) {
184 LOG_ERROR_S << "Worker caught future_error: " << fe.what();
185 } catch (const std::exception& e) {
186 LOG_ERROR_S << "Worker caught exception: " << e.what();
187 } catch (...) {
188 LOG_ERROR_S << "Worker caught unknown exception.";
189 }
190 });
191
192 return res;
193}
