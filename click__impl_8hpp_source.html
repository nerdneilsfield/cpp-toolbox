73// Helper function template to parse the underlying value for std::optional
74// Moved to toolbox::utils namespace to potentially help MSVC name lookup
75template<typename T_Optional> // T_Optional is std::optional<U>
76bool parse_optional_value(const std::string& input, T_Optional& output)
77{
78 // This function is only instantiated when T_Optional is std::optional,
79 // so using detail::optional_value_type_t here is safe.
80 using optional_value_type = detail::optional_value_type_t<T_Optional>;
81 optional_value_type parsed_value;
82 std::stringstream ss(input);
83 bool parse_success = false;
84
85 if constexpr (std::is_same_v<optional_value_type, bool>) {
86 std::string lower_input;
87 std::transform(input.begin(),
88 input.end(),
89 std::back_inserter(lower_input),
90 ::tolower);
91 if (lower_input == "true" || lower_input == "1") {
92 parsed_value = true;
93 parse_success = true;
94 } else if (lower_input == "false" || lower_input == "0") {
95 parsed_value = false;
96 parse_success = true;
97 } else {
98 parse_success = false;
99 }
100 } else if constexpr (std::is_integral_v<optional_value_type>) {
101 if (input.size() > 2 && input[0] == '0'
102 && (input[1] == 'x' || input[1] == 'X')) {
103 ss >> std::hex >> parsed_value;
104 } else {
105 ss >> parsed_value;
106 }
107 parse_success = (ss.eof() && !ss.fail());
108 } else if constexpr (std::is_floating_point_v<optional_value_type>) {
109 ss >> parsed_value;
110 parse_success = (ss.eof() && !ss.fail());
111 } else if constexpr (std::is_same_v<optional_value_type, std::string>) {
112 parsed_value = input;
113 parse_success = true;
114 } else {
115 if constexpr (detail::has_istream_operator_v<optional_value_type>) {
116 ss >> parsed_value;
117 parse_success = (ss.eof() && !ss.fail());
118 } else {
119 LOG_WARN_S << "Unsupported type for default optional parser: " << typeid(optional_value_type).name();
120 parse_success = false;
121 }
122 }
123
124 if (parse_success) {
125 output = parsed_value;
126 } else {
127 output = std::nullopt;
128 }
129 return parse_success;
130}

209template<typename T>
210void option_t<T>::set_default_parser()
211{
212 if constexpr (detail::is_optional_v<T>) {
213 // Lambda specifically for std::optional types
214 parser_ = [](const std::string& input, T& output) -> bool
215 {
216 if (input.empty()) {
217 output = std::nullopt;
218 return true; // Successfully handled empty input
219 }
220 // Delegate actual value parsing to the helper function
221 return parse_optional_value(input, output);
222 };
223 } else {
224 // Lambda specifically for non-optional types
225 parser_ = [](const std::string& input, T& output) -> bool
226 {
227 std::stringstream ss(input);
228 if constexpr (std::is_same_v<T, bool>) {
229 std::string lower_input;
230 std::transform(input.begin(),
231 input.end(),
232 std::back_inserter(lower_input),
233 ::tolower);
234 if (lower_input == "true" || lower_input == "1") {
235 output = true;
236 return true;
237 } else if (lower_input == "false" || lower_input == "0") {
238 output = false;
239 return true;
240 } else {
241 return false; // Invalid boolean string
242 }
243 } else if constexpr (std::is_integral_v<T>) {
244 if (input.size() > 2 && input[0] == '0'
245 && (input[1] == 'x' || input[1] == 'X')) {
246 ss >> std::hex >> output;
247 } else {
248 ss >> output;
249 }
250 } else if constexpr (std::is_floating_point_v<T>) {
251 ss >> output;
252 } else if constexpr (std::is_same_v<T, std::string>) {
253 output = input;
254 return true;
255 } else {
256 if constexpr (detail::has_istream_operator_v<T>) {
257 ss >> output;
258 } else {
259 LOG_WARN_S << "Unsupported type for default parser (non-optional): " << typeid(T).name();
260 return false;
261 }

435 + short_name);

440 option_t<T>* option_ptr = option.get();
