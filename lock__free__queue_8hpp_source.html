178inline void scan_retired_nodes()
179{
180 // 1. 收集所有线程的活跃危险指针/Collect all active hazard pointers from all
181 // threads
182 std::vector<void*> active_hps;
183 {
184 std::lock_guard lock(g_hp_list_mutex); // 保护迭代/Protects iteration
185 for (const HPRec* rec : g_hp_list) {
186 // 检查记录是否真的被占用/Check if record is actually owned
187 if (rec->owner_thread_id.load(std::memory_order_acquire)
188 != std::thread::id()) {
189 for (size_t i = 0; i < MAX_HAZARD_POINTERS_PER_THREAD; ++i) {
190 void* hp = rec->hazard_pointers[i].load(std::memory_order_acquire);
191 if (hp) {
192 active_hps.push_back(hp);
193 }
194 }
195 }
196 }
197 }
198
199 // 2. 检查线程本地退休列表中的每个节点/Check each node in the thread-local
200 // retired list
201 auto it = t_retired_list.begin();
202 while (it != t_retired_list.end()) {
203 void* node_to_check = it->first;
204 const auto& deleter = it->second; // 获取删除器/Get the deleter
205 bool is_hazardous = false;
206 // 检查收集的危险指针/Check against collected HPs
207 for (void* active_hp : active_hps) {
208 if (node_to_check == active_hp) {
209 is_hazardous = true;
210 break;
211 }
212 }
213
214 // 如果在活跃HP中未找到,则可以安全删除/If not found in active HPs, it's safe
215 // to delete
216 if (!is_hazardous) {
217 // 使用存储的删除器安全删除/Safely delete using the stored deleter
218 deleter(node_to_check);
219
220 // 通过与最后一个元素交换并弹出来高效移除元素/Efficiently remove the
221 // element by swapping with the last and popping
222 if (it != t_retired_list.end() - 1) {
223 *it = std::move(t_retired_list.back());
224 }
225 t_retired_list.pop_back();
226 } else {
227 ++it; // 保留到下次扫描/Keep for next scan
228 }
229 }
230}

459 void enqueue(T value)
460 {
461 Node* new_node = new Node(std::move(value));
462
463 while (true) {
464 Node* tail_snapshot = tail_.load(std::memory_order_acquire);
465 Node* next_snapshot = tail_snapshot->next.load(std::memory_order_acquire);
466
467 // 重新检查尾部一致性/Re-check tail consistency
468 if (tail_snapshot == tail_.load(std::memory_order_acquire)) {
469 if (next_snapshot == nullptr) {
470 // 尝试链接新节点/Try to link the new node
471 if (tail_snapshot->next.compare_exchange_weak(
472 next_snapshot,
473 new_node,
474 std::memory_order_release,
475 std::memory_order_relaxed))
476 {
477 // 尝试移动尾指针(可选,尽力而为)
478 // (Try to swing the tail pointer (optional, best effort))
479 tail_.compare_exchange_strong(tail_snapshot,
480 new_node,
481 std::memory_order_release,
482 std::memory_order_relaxed);
483 return; // 入队成功/Enqueue successful
484 }
485 } else {
486 // 尾部已前进,尝试帮助移动尾指针
487 // (Tail already advanced, try to help swing the tail pointer)
488 tail_.compare_exchange_strong(tail_snapshot,
489 next_snapshot,
490 std::memory_order_release,
491 std::memory_order_relaxed);
492 }
493 }

523 bool try_dequeue(T& result)
524 {
525 [[maybe_unused]] detail::HPRec* hp_rec =
526 detail::acquire_hp_record(); // 确保HP记录存在/Ensure HP record exists
527
528 while (true) {
529 // 获取危险指针/Acquire Hazard Pointers
530 Node* head_snapshot = head_.load(std::memory_order_acquire);
531 detail::set_hazard_pointer(0, head_snapshot);
532 // 验证设置HP后head未改变/Verify head hasn't changed after setting HP
533 if (head_snapshot != head_.load(std::memory_order_acquire)) {
534 continue; // Head改变,重试/Head changed, retry
535 }
536
537 Node* next_snapshot = head_snapshot->next.load(std::memory_order_acquire);
538 detail::set_hazard_pointer(1, next_snapshot);
539
540 // 设置HP后重新验证head和next指针
541 // (Re-verify head and next pointers after setting HPs)
542 if (head_snapshot != head_.load(std::memory_order_acquire)) {
543 continue; // Head改变,重试循环将重置HP/Head changed, retry loop will
544 // reset HPs
545 }
546 Node* current_next = head_snapshot->next.load(std::memory_order_acquire);
547 if (next_snapshot != current_next) {
548 continue;
549 }
550
551 Node* tail_snapshot = tail_.load(std::memory_order_acquire);
552
553 if (head_snapshot == tail_snapshot) {
554 // 队列为空或暂时不一致(入队进行中)
555 // (Queue is empty or transiently inconsistent (enqueue in progress))
556 if (next_snapshot == nullptr) {
557 detail::clear_hazard_pointer(
558 0); // 返回前清除HP/Clear HPs before returning
559 detail::clear_hazard_pointer(1);
560 return false; // 队列为空/Queue is empty
561 }
562 // 帮助前进尾部/Help advance tail
563 tail_.compare_exchange_strong(tail_snapshot,
564 next_snapshot,
565 std::memory_order_release,
566 std::memory_order_relaxed);
567 } else {
568 // 队列非空,尝试出队/Queue is not empty, try to dequeue
569 if (next_snapshot == nullptr) {
570 continue;
571 }
572
573 // 尝试前移head指针/Attempt to move the head pointer forward
574 if (head_.compare_exchange_weak(head_snapshot,
575 next_snapshot,
576 std::memory_order_release,
577 std::memory_order_relaxed))
578 {
579 result = std::move(
580 next_snapshot->data); // 数据在新head中/Data is in the NEW head
581
582 // 使用危险指针安全退休旧head节点
583 // (Retire the old head node safely using Hazard Pointers)
584 retire_queue_node(head_snapshot); // 退休旧的哑节点/数据节点/Retire
585 // the old dummy/data node
586
587 // 清除此操作的危险指针/Clear Hazard Pointers for this operation
588 detail::clear_hazard_pointer(0);
589 detail::clear_hazard_pointer(1);
590
591 return true; // 出队成功/Dequeue successful
592 }
593 }
594 }
595 // 清理HP(如果某种情况下未返回就退出循环,不应该发生)
596 // (Cleanup HPs if somehow exited loop without returning (shouldn't happen))
597 detail::clear_hazard_pointer(0);
598 detail::clear_hazard_pointer(1);
599 return false; // 应该不可达/Should be unreachable
600 }
