37template<typename DataType>
38DataType bfknn_parallel_t<DataType>::compute_distance(
39 const point_t<data_type>& p1,
40 const point_t<data_type>& p2,
41 metric_type_t metric) const
42{
43 switch (metric) {
44 case metric_type_t::euclidean: {
45 data_type dx = p1.x - p2.x;
46 data_type dy = p1.y - p2.y;
47 data_type dz = p1.z - p2.z;
48 return std::sqrt(dx * dx + dy * dy + dz * dz);
49 }
50 case metric_type_t::manhattan: {
51 return std::abs(p1.x - p2.x) + std::abs(p1.y - p2.y)
52 + std::abs(p1.z - p2.z);
53 }
54 case metric_type_t::chebyshev: {
55 return std::max({std::abs(p1.x - p2.x),
56 std::abs(p1.y - p2.y),
57 std::abs(p1.z - p2.z)});
58 }
59 case metric_type_t::minkowski: {
60 data_type p = 3.0; // Default p value for Minkowski
61 data_type sum = std::pow(std::abs(p1.x - p2.x), p)
62 + std::pow(std::abs(p1.y - p2.y), p)
63 + std::pow(std::abs(p1.z - p2.z), p);
64 return std::pow(sum, 1.0 / p);
65 }
66 default:
67 return compute_distance(p1, p2, metric_type_t::euclidean);
68 }
69}

109 indices[i] = distance_index_pairs[i].second;

111 } else {
112 // Parallel implementation using thread pool
113 auto& thread_pool = toolbox::base::thread_pool_singleton_t::instance();
114 const std::size_t num_threads = thread_pool.get_thread_count();
115 const std::size_t chunk_size = (cloud_size + num_threads - 1) / num_threads;
116
117 using pair_type = std::pair<data_type, std::size_t>;
118 std::vector<std::vector<pair_type>> thread_results(num_threads);
119 std::vector<std::future<void>> futures;
120
121 // Launch parallel tasks
122 for (std::size_t t = 0; t < num_threads; ++t) {
123 const std::size_t start = t * chunk_size;
124 const std::size_t end = std::min(start + chunk_size, cloud_size);
125
126 if (start >= cloud_size)
127 break;
128
129 futures.push_back(thread_pool.submit(
130 [this, start, end, &query_point, &thread_results, t, num_neighbors]()
131 {
132 std::vector<pair_type> local_pairs;
133 local_pairs.reserve(end - start);
134
135 for (std::size_t i = start; i < end; ++i) {
136 data_type dist =
137 compute_distance(query_point, m_cloud->points[i], m_metric);
138 local_pairs.emplace_back(dist, i);
139 }
140
141 // Partial sort local results
142 std::size_t local_k = std::min(num_neighbors, local_pairs.size());
143 std::partial_sort(local_pairs.begin(),
144 local_pairs.begin() + local_k,
145 local_pairs.end(),
146 [](const auto& a, const auto& b)
147 { return a.first < b.first; });
148
149 // Keep only top k
150 local_pairs.resize(local_k);
151 thread_results[t] = std::move(local_pairs);
152 }));
153 }
154
155 // Wait for all tasks to complete
156 for (auto& future : futures) {
157 future.wait();
158 }
159
160 // Merge results from all threads
161 std::vector<pair_type> all_pairs;
162 for (const auto& thread_result : thread_results) {
163 all_pairs.insert(
164 all_pairs.end(), thread_result.begin(), thread_result.end());
165 }
166
167 // Final sort to get global k nearest
168 std::partial_sort(all_pairs.begin(),
169 all_pairs.begin() + num_neighbors,
170 all_pairs.end(),
171 [](const auto& a, const auto& b)
172 { return a.first < b.first; });
173
174 // Extract results
175 indices.resize(num_neighbors);
176 distances.resize(num_neighbors);
177 for (std::size_t i = 0; i < num_neighbors; ++i) {
178 distances[i] = all_pairs[i].first;
179 indices[i] = all_pairs[i].second;
180 }
181 }
182
183 return true;
