204template<typename DataType>
205std::vector<typename four_pcs_registration_t<DataType>::base_4pcs_t>
206four_pcs_registration_t<DataType>::extract_coplanar_bases(
207 const std::vector<std::size_t>& indices, const point_cloud_ptr& cloud,
208 std::size_t num_bases) const
209{
210 std::vector<base_4pcs_t> bases;
211 bases.reserve(num_bases);
212
213 if (indices.size() < 4) {
214 return bases;
215 }
216
217 std::mt19937 generator(this->get_random_seed());
218 std::uniform_int_distribution<std::size_t> dist(0, indices.size() - 1);
219
220 const DataType coplanar_tolerance = m_delta * 2; // 共面容差 / Coplanarity tolerance
221 std::size_t attempts = 0;
222 const std::size_t max_attempts = num_bases * 100; // 最大尝试次数 / Max attempts
223
224 while (bases.size() < num_bases && attempts < max_attempts) {
225 attempts++;
226
227 // 随机选择4个不同的点 / Randomly select 4 different points
228 std::array<std::size_t, 4> selected_indices;
229 std::unordered_set<std::size_t> used;
230
231 for (std::size_t i = 0; i < 4; ++i) {
232 std::size_t idx;
233 do {
234 idx = dist(generator);
235 } while (used.find(idx) != used.end());
236 used.insert(idx);
237 selected_indices[i] = indices[idx];
238 }
239
240 // 提取点坐标 / Extract point coordinates
241 base_4pcs_t base;
242 base.indices = selected_indices;
243 for (std::size_t i = 0; i < 4; ++i) {
244 const auto& pt = cloud->points[selected_indices[i]];
245 base.points[i] = vector3_t(pt.x, pt.y, pt.z);
246 }
247
248 // 检查是否共面 / Check if coplanar
249 if (!are_coplanar(base.points, coplanar_tolerance)) {
250 continue;
251 }
252
253 // 计算平面参数 / Compute plane parameters
254 vector3_t v1 = base.points[1] - base.points[0];
255 vector3_t v2 = base.points[2] - base.points[0];
256 base.normal = v1.cross(v2).normalized();
257 base.d = -base.normal.dot(base.points[0]);
258
259 // 计算仿射不变量 / Compute affine invariants
260 compute_invariants(base);
261
262 // 检查基的有效性（点不能太近） / Check base validity (points not too close)
263 bool valid = true;
264 for (std::size_t i = 0; i < 4; ++i) {
265 for (std::size_t j = i + 1; j < 4; ++j) {
266 DataType dist = (base.points[i] - base.points[j]).norm();
267 if (dist < m_delta * 10) { // 点间最小距离 / Min distance between points
268 valid = false;
269 break;
270 }
271 }
272 if (!valid) break;
273 }
274
275 if (valid) {
276 bases.push_back(base);
277 }
278 }
279
280 return bases;
281}

283template<typename DataType>
284bool four_pcs_registration_t<DataType>::are_coplanar(const std::array<vector3_t, 4>& points,
285 DataType tolerance) const
286{
287 // 使用前三个点定义平面 / Use first three points to define plane
288 vector3_t v1 = points[1] - points[0];
289 vector3_t v2 = points[2] - points[0];
290 vector3_t normal = v1.cross(v2);
291
292 // 检查是否退化（三点共线） / Check if degenerate (three points collinear)
293 if (normal.norm() < std::numeric_limits<DataType>::epsilon()) {
294 return false;
295 }
296
297 normal.normalize();
298 DataType d = -normal.dot(points[0]);
299
300 // 检查第四个点到平面的距离 / Check distance of fourth point to plane
301 DataType dist = std::abs(normal.dot(points[3]) + d);
302
303 return dist <= tolerance;
304}

306template<typename DataType>
307void four_pcs_registration_t<DataType>::compute_invariants(base_4pcs_t& base) const
308{
309 // 4PCS使用两个交点的比率作为仿射不变量 / 4PCS uses ratios of intersection points as affine invariants
310 // 对角线AC和BD的交点 / Intersection of diagonals AC and BD
311
312 // 计算对角线AC: P0-P2 / Compute diagonal AC: P0-P2
313 vector3_t ac_dir = base.points[2] - base.points[0];
314 DataType ac_len = ac_dir.norm();
315
316 // 计算对角线BD: P1-P3 / Compute diagonal BD: P1-P3
317 vector3_t bd_dir = base.points[3] - base.points[1];
318 DataType bd_len = bd_dir.norm();
319
320 // 计算交点（使用参数方程） / Compute intersection (using parametric equations)
321 // P0 + s * AC_dir = P1 + t * BD_dir
322
323 // 构建线性系统 / Build linear system
324 matrix3_t A;
325 A.col(0) = ac_dir;
326 A.col(1) = -bd_dir;
327 A.col(2) = base.normal; // 添加法向量确保系统非奇异 / Add normal to ensure non-singular
328
329 vector3_t b = base.points[1] - base.points[0];
330
331 // 求解 / Solve
332 vector3_t params = A.fullPivLu().solve(b);
333 DataType s = params[0];
334 DataType t = params[1];
335
336 // 计算不变量（交点在对角线上的位置比率） / Compute invariants (position ratios on diagonals)
337 base.invariant1 = s; // 交点在AC上的位置 / Position on AC
338 base.invariant2 = t; // 交点在BD上的位置 / Position on BD
339}

341template<typename DataType>
342std::vector<typename four_pcs_registration_t<DataType>::base_4pcs_t>
343four_pcs_registration_t<DataType>::find_congruent_sets(
344 const base_4pcs_t& source_base, const std::vector<std::size_t>& target_indices,
345 const point_cloud_ptr& target_cloud) const
346{
347 std::vector<base_4pcs_t> congruent_bases;
348
349 // 容差 / Tolerances
350 const DataType invariant_tolerance = 0.1; // 不变量容差 / Invariant tolerance
351 const DataType distance_tolerance = m_delta * 2; // 距离容差 / Distance tolerance
352
353 // 计算源基的边长 / Compute source base edge lengths
354 std::array<DataType, 6> source_distances; // 6条边 / 6 edges
355 std::size_t edge_idx = 0;
356 for (std::size_t i = 0; i < 4; ++i) {
357 for (std::size_t j = i + 1; j < 4; ++j) {
358 source_distances[edge_idx++] = (source_base.points[i] - source_base.points[j]).norm();
359 }
360 }
361
362 // 寻找所有可能的匹配 / Find all possible matches
363 // 这里简化实现，实际4PCS使用更复杂的搜索策略 / Simplified implementation, actual 4PCS uses more complex search
364
365 if (target_indices.size() < 4) {
366 return congruent_bases;
367 }
368
369 // 随机尝试一些组合 / Try some random combinations
370 std::mt19937 generator(this->get_random_seed());
371 std::uniform_int_distribution<std::size_t> dist(0, target_indices.size() - 1);
372
373 const std::size_t max_tries = 1000;
374 for (std::size_t try_idx = 0; try_idx < max_tries; ++try_idx) {
375 // 随机选择4个目标点 / Randomly select 4 target points
376 std::array<std::size_t, 4> selected_indices;
377 std::unordered_set<std::size_t> used;
378
379 bool valid_selection = true;
380 for (std::size_t i = 0; i < 4; ++i) {
381 std::size_t idx;
382 std::size_t attempts = 0;
383 do {
384 idx = dist(generator);
385 attempts++;
386 if (attempts > 100) {
387 valid_selection = false;
388 break;
389 }
390 } while (used.find(idx) != used.end());
391
392 if (!valid_selection) break;
393 used.insert(idx);
394 selected_indices[i] = target_indices[idx];
395 }
396
397 if (!valid_selection) continue;
398
399 // 构建目标基 / Build target base
400 base_4pcs_t target_base;
401 target_base.indices = selected_indices;
402 for (std::size_t i = 0; i < 4; ++i) {
403 const auto& pt = target_cloud->points[selected_indices[i]];
404 target_base.points[i] = vector3_t(pt.x, pt.y, pt.z);
405 }
406
407 // 检查是否共面 / Check if coplanar
408 if (!are_coplanar(target_base.points, distance_tolerance)) {
409 continue;
410 }
411
412 // 计算目标基的边长 / Compute target base edge lengths
413 std::array<DataType, 6> target_distances;
414 edge_idx = 0;
415 for (std::size_t i = 0; i < 4; ++i) {
416 for (std::size_t j = i + 1; j < 4; ++j) {
417 target_distances[edge_idx++] = (target_base.points[i] - target_base.points[j]).norm();
418 }
419 }
420
421 // 检查边长是否匹配（允许不同的点对应关系） / Check if edge lengths match (allow different correspondences)
422 bool distances_match = true;
423 for (std::size_t i = 0; i < 6; ++i) {
424 bool found_match = false;
425 for (std::size_t j = 0; j < 6; ++j) {
426 if (std::abs(source_distances[i] - target_distances[j]) < distance_tolerance) {
427 found_match = true;
428 break;
429 }
430 }
431 if (!found_match) {
432 distances_match = false;
433 break;
434 }
435 }
436
437 if (!distances_match) {
438 continue;
439 }
440
441 // 计算不变量 / Compute invariants
442 compute_invariants(target_base);
443
444 // 检查不变量是否匹配 / Check if invariants match
445 if (std::abs(source_base.invariant1 - target_base.invariant1) < invariant_tolerance &&
446 std::abs(source_base.invariant2 - target_base.invariant2) < invariant_tolerance) {
447 congruent_bases.push_back(target_base);
448 }
449 }
450
451 return congruent_bases;
452}

454template<typename DataType>
455typename four_pcs_registration_t<DataType>::transformation_t
456four_pcs_registration_t<DataType>::estimate_transformation(const base_4pcs_t& source_base,
457 const base_4pcs_t& target_base) const
458{
459 transformation_t transform;
460 transform.setIdentity();
461
462 // 使用SVD估计变换（类似RANSAC） / Estimate transformation using SVD (similar to RANSAC)
463
464 // 计算质心 / Compute centroids
465 vector3_t source_centroid = vector3_t::Zero();
466 vector3_t target_centroid = vector3_t::Zero();
467
468 for (std::size_t i = 0; i < 4; ++i) {
469 source_centroid += source_base.points[i];
470 target_centroid += target_base.points[i];
471 }
472
473 source_centroid /= 4;
474 target_centroid /= 4;
475
476 // 构建协方差矩阵 / Build covariance matrix
477 matrix3_t H = matrix3_t::Zero();
478
479 for (std::size_t i = 0; i < 4; ++i) {
480 vector3_t src_centered = source_base.points[i] - source_centroid;
481 vector3_t tgt_centered = target_base.points[i] - target_centroid;
482 H += src_centered * tgt_centered.transpose();
483 }
484
485 // SVD分解 / SVD decomposition
486 Eigen::JacobiSVD<matrix3_t> svd(H, Eigen::ComputeFullU | Eigen::ComputeFullV);
487 matrix3_t U = svd.matrixU();
488 matrix3_t V = svd.matrixV();
489
490 // 计算旋转矩阵 / Compute rotation matrix
491 matrix3_t R = V * U.transpose();
492
493 // 处理反射情况 / Handle reflection case
494 if (R.determinant() < 0) {
495 V.col(2) *= -1;
496 R = V * U.transpose();
497 }
498
499 // 计算平移向量 / Compute translation vector
500 vector3_t t = target_centroid - R * source_centroid;
501
502 // 构建4x4变换矩阵 / Build 4x4 transformation matrix
503 transform.template block<3, 3>(0, 0) = R;
504 transform.template block<3, 1>(0, 3) = t;
505
506 return transform;
507}

509template<typename DataType>
510DataType four_pcs_registration_t<DataType>::compute_lcp_score(
511 const transformation_t& transform, std::vector<std::size_t>& inliers) const
512{
513 inliers.clear();
514
515 auto source_cloud = this->get_source_cloud();
516 const DataType threshold_squared = m_delta * m_delta;
517
518 // 使用采样点计算LCP / Compute LCP using sampled points
519 DataType total_distance = 0;
520
521 for (std::size_t src_idx : m_source_samples) {
522 const auto& src_pt = source_cloud->points[src_idx];
523
524 // 变换源点 / Transform source point
525 vector3_t src_vec(src_pt.x, src_pt.y, src_pt.z);
526 vector3_t transformed = transform.template block<3, 3>(0, 0) * src_vec +
527 transform.template block<3, 1>(0, 3);
528
529 // 在目标点云中查找最近邻 / Find nearest neighbor in target cloud
530 toolbox::types::point_t<DataType> query_pt;
531 query_pt.x = transformed[0];
532 query_pt.y = transformed[1];
533 query_pt.z = transformed[2];
534
535 std::vector<std::size_t> indices;
536 std::vector<DataType> distances;
537 m_target_kdtree->kneighbors(query_pt, 1, indices, distances);
538
539 if (!indices.empty() && distances[0] <= threshold_squared) {
540 inliers.push_back(src_idx);
541 total_distance += std::sqrt(distances[0]);
542 }
543 }
544
545 // LCP评分：平均内点距离 / LCP score: average inlier distance
546 if (inliers.empty()) {
547 return std::numeric_limits<DataType>::max();
548 }
549
550 return total_distance / inliers.size();
551}

564template<typename DataType>
565bool four_pcs_registration_t<DataType>::is_valid_transformation(
566 const transformation_t& transform) const
567{
568 // 检查旋转部分是否有效 / Check if rotation part is valid
569 matrix3_t R = transform.template block<3, 3>(0, 0);
570
571 // 检查行列式（应该接近1） / Check determinant (should be close to 1)
572 DataType det = R.determinant();
573 if (std::abs(det - 1.0) > 0.1) {
574 return false;
575 }
576
577 // 检查正交性 / Check orthogonality
578 matrix3_t should_be_identity = R * R.transpose();
579 matrix3_t identity = matrix3_t::Identity();
580 DataType error = (should_be_identity - identity).norm();
581
582 return error < 0.1;
583}
